## 1. 栈

### 1.1 实现

```js
class Stack {
  constructor(...arr) {
    this.stack = [...arr]
    this.size = arr.length
  }
  
  isEmpty() {
    return this.size === 0? true: false
  }

  push(...args) {
    this.stack.push(...args)
    this.size += args.length
    return this.size
  }

  pop() {
    if (this.isEmpty()) {
      return
    }
    this.size--
    return this.stack.pop()
  }

  peak() {
    if(this.isEmpty()) {
      return
    }
    return this.stack[this.size - 1]
  }

  toString() {
    return this.stack
  }
}
```

### 1.2 应用

#### 1.2.1 判断括号是否匹配

```js
function isValid(str) {
  const rules = {
      '{': '}',
      '(': ')',
      '[': ']'
    },
    stack = new Stack()
  for (const c of str) {
    if (stack.size === 0) {
      stack.push(c)
      continue
    }
    if (rules[stack.peak()] === c) {
      stack.pop()
    } else {
      stack.push(c)
    }

  }
  return stack.size === 0
}
```

#### 1.2.2 使用栈模拟队列

使用两个栈可以模拟队列

```js
class QueueLike {
  constructor(...args) {
    this.masterStack = new Stack(...args)
    this.helperStack = new Stack()
  }

  enqueue(...args) {
    this.masterStack.push(...args)
  }

  dequeue() {
    if (this.masterStack.isEmpty()) {
      return
    }
    for (let i = 0, length = this.masterStack.size; i < length - 1; i++) {
      this.helperStack.push(this.masterStack.pop())
    }
    const result = this.masterStack.pop()
    for (let i = 0, length = this.helperStack.size; i < length; i++) {
      this.masterStack.push(this.helperStack.pop())
    }
    return result
  }

  toString() {
    return this.masterStack.toString()
  }
}
```

## 2. 队列

### 2.1 实现

```js
class Queue {
  constructor(...args) {
    this.queue = [...args]
  }

  get size() {
    return this.queue.length
  }

  isEmpty() {
    return this.size === 0
  }

  enqueue(...value) {
    this.queue.push(...value)
    return this.size
  }

  dequeue() {
    if (this.isEmpty()) {
      return
    }
    return this.queue.shift()
  }

  toString() {
    return this.queue.toString()
  }
}
```

### 2.2 应用

#### 2.2.1 使用队列模拟栈

```js
class StackLike {
  constructor(...args) {
    this.queue = new Queue(...args)
  }

  get size() {
    return this.queue.size
  }

  isEmpty() {
    return this.size === 0
  }

  push(...args) {
    this.queue.enqueue(...args)
    return this.size
  }

  pop() {
    for(let i = 0, length = this.size; i < length - 1; i++) {
      this.queue.enqueue(this.queue.dequeue())
    }
    return this.queue.dequeue()
  }

  toString() {
    return this.queue.toString()
  }
}
```
